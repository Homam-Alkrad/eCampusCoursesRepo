#include <iostream>
#include <string>
using namespace std;

struct Type {
    string quiz;
    string mid;
    string final;
};

struct Exam {
    string name;
    Type type;
    double mark;
};

struct Student {
    int id;
    string name;
    Exam exam;
};

void printStudents(Student* arr, int size) {
    cout << "\n=== Student Records ===" << endl;
    for (int i = 0; i < size; i++) {
        cout << "Student ID: " << (arr + i)->id << endl;
        cout << "Name: " << (arr + i)->name << endl;
        cout << "Exam: " << (arr + i)->exam.name << endl;
        cout << "Type: " << (arr + i)->exam.type.mid << endl;
        cout << "Mark: " << (arr + i)->exam.mark << endl;
        cout << "-------------------" << endl;
    }
}

void searchStudent(Student* arr, int size, int searchId) {
    bool found = false;
    for (int i = 0; i < size; i++) {
        if ((arr + i)->id == searchId) {
            cout << "\n=== Student Found ===" << endl;
            cout << "ID: " << (arr + i)->id << endl;
            cout << "Name: " << (arr + i)->name << endl;
            cout << "Mark: " << (arr + i)->exam.mark << endl;
            found = true;
            break;
        }
    }
    if (!found) {
        cout << "Student with ID " << searchId << " not found." << endl;
    }
}

void findMaxMark(Student* arr, int size) {
    double maxMark = (arr + 0)->exam.mark;
    int maxIndex = 0;
    
    for (int i = 1; i < size; i++) {
        if ((arr + i)->exam.mark > maxMark) {
            maxMark = (arr + i)->exam.mark;
            maxIndex = i;
        }
    }
    
    cout << "\n=== Student with Maximum Mark ===" << endl;
    cout << "Name: " << (arr + maxIndex)->name << endl;
    cout << "Mark: " << maxMark << endl;
}

class Node {

        cout << "Node constructor called with value: " << value << endl;

    }

        }
    }

#include <iostream>
using namespace std;

class Node {
public:
    int data;
    Node* next;
    
    Node(int value) {
        data = value;
        next = nullptr;
    }
};

class LinkedList {
private:
    Node* head;
    
public:
    LinkedList() {
        cout << "LinkedList constructor called" << endl;
        head = nullptr;
    }    
    bool isEmpty() {
        return head == nullptr;
    }
    
    Node* getHead() {
        return head;
    }
    
    void add(int value) {
        Node* newNode = new Node(value);
        
        if (head == nullptr) {
            head = newNode;
            cout << "Added first node with value: " << value << endl;
        } else {
            Node* current = head;
            while (current->next != nullptr) {
                current = current->next;
            }
            current->next = newNode;
            cout << "Added node with value: " << value << endl;
        }
    }
    
    void display() {
        if (head == nullptr) {
            cout << "List is empty" << endl;
            return;
        }
        
        Node* current = head;
        cout << "List: ";
        while (current != nullptr) {
            cout << current->data;
            if (current->next != nullptr) {
                cout << " -> ";
            }
            current = current->next;
        }
        cout << endl;
    }
};

int main() {
    cout << "=== Linked List Class Demo ===" << endl;
    
    LinkedList list;
    
    cout << "\nChecking if list is empty:" << endl;

#include <iostream>

class Node {

    int data;
    Node* next;
    
    Node(int value) {
        data = value;
        next = nullptr;
    }
};

class LinkedList {
private:
    Node* head;
    
public:
    LinkedList() {
        head = nullptr;
    }
    
    void add(int value) {
        Node* newNode = new Node(value);
        
        if (head == nullptr) {
            head = newNode;
        } else {
            Node* current = head;
            while (current->next != nullptr) {
                current = current->next;
            }
            current->next = newNode;
        }
    }
    
    void display() {
        if (head == nullptr) {
            cout << "List is empty" << endl;
            return;
        }
        
        Node* current = head;
        cout << "List: ";
        while (current != nullptr) {
            cout << current->data;
            if (current->next != nullptr) {
                cout << " -> ";
            }
            current = current->next;
        }
        cout << endl;
    }
    
    bool search(int value) {
        Node* current = head;
        int position = 0;
        
        while (current != nullptr) {
            if (current->data == value) {
                cout << "Value " << value << " found at position " << position << endl;
                return true;
            }
            current = current->next;
            position++;
        }
        
        cout << "Value " << value << " not found in the list" << endl;
        return false;
    }
    
    int sum() {
        int total = 0;
        Node* current = head;
        
        while (current != nullptr) {
            total += current->data;
            current = current->next;
        }
        
        return total;
    }
    
    int count() {
        int counter = 0;
        Node* current = head;
        
        while (current != nullptr) {
            counter++;
            current = current->next;
        }
        
        return counter;
    }
    
    double average() {
        int total = sum();
        int counter = count();
        
        if (counter == 0) {
            return 0.0;
        }
        
        return (double)total / counter;
    }
    
    int findMax() {
        if (head == nullptr) {
            cout << "List is empty" << endl;
            return -1;
        }
        
        int max = head->data;
        Node* current = head->next;
        
        while (current != nullptr) {
            if (current->data > max) {
                max = current->data;
            }
            current = current->next;
        }
        
        return max;
    }
    
    int findMin() {
        if (head == nullptr) {
            cout << "List is empty" << endl;
            return -1;
        }
        
        int min = head->data;
        Node* current = head->next;
        
        while (current != nullptr) {
            if (current->data < min) {
                min = current->data;
            }
            current = current->next;
        }
        
        return min;
    }
};

#include <iostream>

class Node {

    Node* next;
    
    Node(int value) {
        data = value;
        next = nullptr;
    }

class LinkedList {
private:

public:
    LinkedList() {
        head = nullptr;
    }
    
    void addAtFirst(int value) {
        Node* newNode = new Node(value);
        newNode->next = head;
        head = newNode;
        cout << "Added " << value << " at first position" << endl;
    }
    
    void addAtLast(int value) {
        Node* newNode = new Node(value);
        
        if (head == nullptr) {
            head = newNode;
            cout << "Added " << value << " as first element" << endl;
            return;
        }
        
        Node* current = head;
        while (current->next != nullptr) {
            current = current->next;
        }
        current->next = newNode;
        cout << "Added " << value << " at last position" << endl;
    }
    
    void addAtEnd(int value) {
        addAtLast(value);
    }
    
    void addAtPosition(int value, int position) {
        if (position < 0) {
            cout << "Invalid position" << endl;
            return;
        }
        
        if (position == 0) {
            addAtFirst(value);
            return;
        }
        
        Node* newNode = new Node(value);
        Node* current = head;
        
        for (int i = 0; i < position - 1 && current != nullptr; i++) {
            current = current->next;
        }
        
        if (current == nullptr) {
            cout << "Position " << position << " is out of bounds" << endl;
            delete newNode;
            return;
        }
        
        newNode->next = current->next;
        current->next = newNode;
        cout << "Added " << value << " at position " << position << endl;
    }
    
    void addAfterValue(int newValue, int afterValue) {
        Node* current = head;

    }
